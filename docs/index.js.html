/**

возвращает строку вида "S = 120"

@param {side} a - First side
@param {side} b - Second side
@param {side} c - Third side

*/

function square(a, b, c){
     let p = (a + b + c) / 2;
     return 'S = ${(p * (p - a) * (p - b) * (p - c)) ** (1/2)}';
}
